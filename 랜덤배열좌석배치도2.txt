    l = Math.floor(Math.random() * number.length/2)*2;
    result = result.concat(number.substr(l,2));
    number = number.substr(0,l).concat(number.substr(l+2,number.length)); 
    document.write(number+"<br>");   
}
document.write(result);


    document.write("<h1>좌석배치도</h1>");
    document.write("<table border = 2 width = 50%");

    for(var i = 0; i < 4; i++) {
    	document.write("<tr>");

    	document.write("<td>"+ result.substr(k,2) +"</td>");
	k=k+2;

	    document.write("<td>"+ result.substr(k,2) +"</td>");
	    k=k+2;
	    
        }
    	document.write("</tr>");
    }
    document.write("</table>");
